audio := Input Resample="yes" SampleRate="22050"
nAudioIn := FrameTokenizer blockSize="512" stepSize="256"
audio > nAudioIn

nFFT := FFT FFTLength="512" FFTWindow="Hanning"
nAudioIn > nFFT

nFFTAbs := Abs
nFFT > nFFTAbs

nMelFilterBank := MelFilterBank MelMaxFreq="6854.0" MelMinFreq="130.0" MelNbFilters="40"
nFFTAbs > nMelFilterBank

nCepstrum := Cepstrum CepsIgnoreFirstCoeff="1" CepsNbCoeffs="13"
nMelFilterBank > nCepstrum

MFCC := Output normalize="-1" version="v0.6" yaafedefinition="MFCC blockSize=512 stepSize=256"
nCepstrum > MFCC

nDMFCC := Derivate DO1Len="4" DO2Len="1" DOrder="1"
Delta_MFCC := Output normalize="-1" version="v0.6" yaafedefinition="MFCC blockSize=512 stepSize=256 > Derivate DOrder=1"
nCepstrum > nDMFCC
nDMFCC > Delta_MFCC

nDDMFCC := Derivate DO1Len="4" DO2Len="1" DOrder="2"
Delta_Delta_MFCC := Output normalize="-1" version="v0.6" yaafedefinition="MFCC blockSize=512 stepSize=256 > Derivate DOrder=2"
nCepstrum > nDDMFCC
nDDMFCC > Delta_Delta_MFCC

nFFTAbsSqr := Sqr
nFFTAbs > nFFTAbsSqr

nOBSI := OBSI
OctaveBandsSignalIntensity := Output normalize="-1" version="v0.6" yaafedefinition="OBSI blockSize=512 stepSize=256"
nFFTAbsSqr > nOBSI
nOBSI > OctaveBandsSignalIntensity

nEnvelope := Envelope 
Envelope := Output normalize="-1" version="v0.6" yaafedefinition="Envelope  EnDecim=200 blockSize=512  stepSize=256"
nAudioIn > nEnvelope
nEnvelope > Envelope

nAutoCorrEnv := AutoCorrelation
Envelope_AutoCorrelation := Output normalize="-1" version="v0.6" yaafedefinition="AutoCorrelation ACNbCoeffs=49  blockSize=512  stepSize=256"
nEnvelope > nAutoCorrEnv
nAutoCorrEnv > Envelope_AutoCorrelation

nRMS := RMS
Energy := Output normalize="-1" version="v0.6" yaafedefinition="Energy blockSize=512 stepSize=256"
nAudioIn > nRMS
nRMS > Energy

nSpectralFlatness := Flatness
Spectral_Flatness := Output normalize="-1" version="v0.6" yaafedefinition="SpectralFlatness blockSize=512 stepSize=256"
nFFTAbs > nSpectralFlatness
nSpectralFlatness > Spectral_Flatness
