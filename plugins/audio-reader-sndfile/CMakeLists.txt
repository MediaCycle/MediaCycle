SET(MEDIA_TYPE "audio")
SET(PLUGIN_NAME "audio-reader-sndfile")
SET(MC_MEDIA_LIBS "${MEDIA_TYPE}-core")
list(APPEND 3RDPARTY_NAMES "libsndfile")
list(APPEND 3RDPARTY_LICENSES "LGPL or GPL")
list(APPEND 3RDPARTY_URLS "http://www.mega-nerd.com/libsndfile")
IF(SUPPORT_AUDIO AND SNDFILE_FOUND)
	INCLUDE (${CMAKE_SOURCE_DIR}/CMakeModules/AddMediaCyclePlugin.cmake)
	ADD_MC_PLUGIN_LIBRARY(${PLUGIN_NAME})
	TARGET_LINK_LIBRARIES(${TARGET_NAME} ${SNDFILE_LIBRARY})
	MESSAGE("[X] ${TARGET_NAME}")
ELSE()
	MESSAGE("[ ] ${PLUGIN_PREFIX}${PLUGIN_NAME}")
ENDIF()
