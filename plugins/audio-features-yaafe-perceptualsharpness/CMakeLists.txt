SET(MEDIA_TYPE "audio")
SET(PLUGIN_NAME "audio-features-yaafe-perceptualsharpness")
SET(MC_MEDIA_LIBS "${MEDIA_TYPE}-core;${MEDIA_TYPE}-features-yaafe")
list(APPEND 3RDPARTY_NAMES "Yet another audio features extraction (Yaafe)")
list(APPEND 3RDPARTY_LICENSES "LGPL")
list(APPEND 3RDPARTY_URLS "http://yaafe.sourceforge.net")
IF(SUPPORT_AUDIO AND USE_YAAFE)
        INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/3rdparty/yaafe)
        INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/3rdparty/yaafe/src_cpp)

	INCLUDE (${CMAKE_SOURCE_DIR}/cmake/AddMediaCyclePlugin.cmake)
	ADD_MC_PLUGIN_LIBRARY(${PLUGIN_NAME})

	IF(WIN32 AND MINGW)
		FILE(GLOB YAAFE_SETTINGS_FILES *.txt)
		foreach(YAAFE_SETTINGS_FILE ${YAAFE_SETTINGS_FILES})
			EXECUTE_PROCESS(COMMAND cp "${YAAFE_SETTINGS_FILE}" "${CMAKE_CURRENT_BINARY_DIR}")
		endforeach(YAAFE_SETTINGS_FILE)
	ENDIF()

	MESSAGE("[X] ${TARGET_NAME}")
ELSE()
	MESSAGE("[ ] ${PLUGIN_PREFIX}${PLUGIN_NAME}")
ENDIF()
