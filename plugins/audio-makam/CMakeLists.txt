SET(MEDIA_TYPE "audio")
SET(PLUGIN_NAME "audio-makam")
SET(MC_3RDPARTIES "octave_embed")
SET(MC_MEDIA_LIBS "${MEDIA_TYPE}-core")
list(APPEND 3RDPARTY_NAMES "Octave")
list(APPEND 3RDPARTY_LICENSES "GPL")
list(APPEND 3RDPARTY_URLS "http://www.octave.org")
list(APPEND 3RDPARTY_NAMES "Makam Toolbox, from Baris Bozkurt et al, ported to Octave by Onur Babacan")

list(APPEND 3RDPARTY_URLS "ftp://ftp.iyte.edu.tr/share/ktm-nota/TuningMeasurement.html")
IF(SUPPORT_AUDIO AND OCTAVE_FOUND AND USE_OCTAVE AND USE_MAKAM)
	INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/3rparty/octave_makam)
	INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/3rparty/octave_yin)
	INCLUDE (${CMAKE_SOURCE_DIR}/CMakeModules/AddMediaCyclePlugin.cmake)
	ADD_MC_PLUGIN_LIBRARY(${PLUGIN_NAME})
	TARGET_LINK_LIBRARIES(${TARGET_NAME} ${OCTAVE_LIBRARY})
	ADD_DEPENDENCIES(${TARGET_NAME} octave_yin)
	MESSAGE("[X] ${TARGET_NAME}")
ELSE()
	MESSAGE("[ ] ${PLUGIN_PREFIX}${PLUGIN_NAME}")
ENDIF()
