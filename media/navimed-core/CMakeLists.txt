SET(MEDIA_TYPE "navimed")
SET(LIBRARY_NAME "mediacycle-${MEDIA_TYPE}-core")
IF(SUPPORT_NAVIMED)
	FILE(GLOB SRC *.cpp *.c)
	FILE(GLOB HDR *.hpp *.h)

	INCLUDE_DIRECTORIES(../image-core ../sensor-core ../text-core)
	ADD_LIBRARY(${LIBRARY_NAME} ${SRC} ${HDR})
	ADD_DEPENDENCIES(${LIBRARY_NAME} mediacycle-image-core mediacycle-text-core mediacycle-sensor-core)
	TARGET_LINK_LIBRARIES(${LIBRARY_NAME} mediacycle-image-core mediacycle-sensor-core mediacycle-text-core)
	IF(TINYXML_FOUND)
		TARGET_LINK_LIBRARIES(${LIBRARY_NAME} ${TINYXML_LIBRARY})
	ELSE()
		ADD_DEPENDENCIES(${LIBRARY_NAME} tinyxml)
		TARGET_LINK_LIBRARIES(${LIBRARY_NAME} tinyxml)
	ENDIF()
	SET_TARGET_PROPERTIES(${LIBRARY_NAME} PROPERTIES FOLDER "${FOLDERNAME}")


	MESSAGE("[X] ${LIBRARY_NAME}")
ELSE()
	MESSAGE("[ ] ${LIBRARY_NAME}")
ENDIF()
