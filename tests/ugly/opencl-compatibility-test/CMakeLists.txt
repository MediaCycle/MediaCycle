SET(TARGET_NAME "opencl-compatibility-test")
FIND_PACKAGE(OpenCL)
IF(OPENCL_FOUND)
	FILE(GLOB SRC *.cpp *.c)
	FILE(GLOB HDR *.hpp *.h)
	INCLUDE_DIRECTORIES(${OPENCL_INCLUDE_DIRS})
	ADD_EXECUTABLE(${TARGET_NAME} ${SRC} ${HDR})
	TARGET_LINK_LIBRARIES(${TARGET_NAME} ${OPENCL_LIBRARIES})
	SET_TARGET_PROPERTIES(${TARGET_NAME} PROPERTIES FOLDER "${FOLDERNAME}")
        MESSAGE("[X] ${TARGET_NAME}")
ELSE()
        MESSAGE("[ ] ${TARGET_NAME}")
ENDIF()
